DROP TABLE DEPT;
DROP TABLE EMP;
DROP TABLE HOSPITAL;
DROP TABLE SALA;
DROP TABLE DOCTOR;
DROP TABLE PLANTILLA;
DROP TABLE ENFERMO;
DROP TABLE OCUPACION;

CREATE TABLE DEPT
  (DEPT_NO number(9,0)
  ,DNOMBRE VARCHAR2(50)
  ,LOC VARCHAR2(50)
);

CREATE TABLE EMP
  (EMP_NO number(9,0)
  ,APELLIDO VARCHAR2(50)
  ,OFICIO VARCHAR2(50)
  ,DIR number(9,0)
  ,FECHA_ALT DATE
  ,SALARIO number(9,0)
  ,COMISION number(9,0)
  ,DEPT_NO number(9,0)
);

CREATE TABLE HOSPITAL(
  HOSPITAL_COD number(9,0)
  ,NOMBRE VARCHAR2(50)
  ,DIRECCION VARCHAR2(50),
  TELEFONO VARCHAR2(9),
  NUM_CAMA number(9,0)
);

CREATE TABLE SALA(
  HOSPITAL_COD number(9,0)
  ,SALA_COD number(9,0)
  ,NOMBRE VARCHAR2(50)
  ,NUM_CAMA number(9,0)
);

CREATE TABLE DOCTOR(
  HOSPITAL_COD number(9,0)
  ,DOCTOR_NO number(9,0)
  ,APELLIDO VARCHAR2(50)
  ,ESPECIALIDAD VARCHAR(50),
  SALARIO number(9,0)
);

CREATE TABLE PLANTILLA(
  HOSPITAL_COD number(9,0)
  ,SALA_COD number(9,0)
  ,EMPLEADO_NO number(9,0)
  ,APELLIDO VARCHAR2(50)
  ,FUNCION VARCHAR2(50)
  ,TURNO VARCHAR2(1)
  ,SALARIO number(9,0)
);

CREATE TABLE ENFERMO(
  INSCRIPCION number(9,0)
  ,APELLIDO VARCHAR2(50)
  ,DIRECCION VARCHAR2(50)
  ,FECHA_NAC DATE
  ,SEXO VARCHAR2(1)
  ,NSS number(9,0)
);

CREATE TABLE OCUPACION(
  INSCRIPCION number(9,0)
  ,HOSPITAL_COD number(9,0)
  ,SALA_COD number(9,0)
  ,CAMA number(9,0)
);

ALTER TABLE EMP
ADD CONSTRAINT PK_EMP
PRIMARY KEY (EMP_NO);

ALTER TABLE DEPT
ADD CONSTRAINT PK_DEPT
PRIMARY KEY (DEPT_NO);

ALTER TABLE HOSPITAL
ADD CONSTRAINT PK_HOSPITAL
PRIMARY KEY (HOSPITAL_COD);

ALTER TABLE DOCTOR
ADD CONSTRAINT PK_DOCTOR
PRIMARY KEY (DOCTOR_NO);

ALTER TABLE PLANTILLA
ADD CONSTRAINT PK_PLANTILLA
PRIMARY KEY (EMPLEADO_NO);

ALTER TABLE ENFERMO
ADD CONSTRAINT PK_ENFERMO
PRIMARY KEY (INSCRIPCION);
