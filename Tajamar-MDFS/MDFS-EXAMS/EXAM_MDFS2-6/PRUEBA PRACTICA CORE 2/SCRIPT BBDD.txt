DROP TABLE GENEROS
DROP TABLE LIBROS
DROP TABLE USUARIOS
DROP TABLE PEDIDOS
DROP VIEW VISTAPEDIDOS
CREATE TABLE GENEROS
(	IdGenero int NOT NULL primary key,
	Nombre nvarchar(50) NULL,
)
GO
CREATE TABLE LIBROS
(
	IdLibro int NOT NULL PRIMARY KEY,
	Titulo nvarchar(50) NULL,
	Autor nvarchar(50) NULL,
	Editorial nvarchar(50) NULL,
	Portada nvarchar(50) NULL,
	Resumen nvarchar(max) NULL,
	Precio int NULL,
	IdGenero int NULL
)
GO
CREATE TABLE USUARIOS
(
	IdUsuario int NOT NULL PRIMARY KEY,
	Nombre nvarchar(50) NULL,
	Apellidos nvarchar(50) NULL,
	Email nvarchar(50) NULL,
	Pass nvarchar(50) NULL,
	Foto nvarchar(50) NULL
)
GO
CREATE TABLE PEDIDOS
(IDPEDIDO INT NOT NULL PRIMARY KEY
, IDFACTURA INT --VALOR REPETIDO PARA LOCALIZAR PEDIDOS
, FECHA DATE
, IDLIBRO INT
, IDUSUARIO INT
, CANTIDAD INT)
GO
CREATE VIEW VISTAPEDIDOS
AS
	SELECT 
	ISNULL(ROW_NUMBER() OVER (ORDER BY USUARIOS.NOMBRE), 1)
	AS IDVISTAPEDIDOS,
	USUARIOS.IdUsuario, 
	USUARIOS.Nombre, USUARIOS.Apellidos
	, LIBROS.Titulo, LIBROS.Precio, LIBROS.Portada
	, PEDIDOS.FECHA, 
LIBROS.Precio * PEDIDOS.Cantidad as
PRECIOFINAL
	FROM USUARIOS
	INNER JOIN PEDIDOS
	ON USUARIOS.IDUSUARIO = PEDIDOS.IDUSUARIO
	INNER JOIN LIBROS
	ON LIBROS.IDLIBRO = PEDIDOS.IDLIBRO
GO
